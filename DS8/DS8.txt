import seaborn as sns
import matplotlib.pyplot as plt

# Load the Titanic dataset
titanic = sns.load_dataset('titanic')

# 1. Histogram of Ticket Fare
plt.figure(figsize=(10, 6))
sns.histplot(titanic['fare'], kde=True, bins=30, color='skyblue')
plt.title('Distribution of Ticket Fare on Titanic')
plt.xlabel('Fare')
plt.ylabel('Frequency')
plt.show()

# 2. Pairplot of Age, Fare, and Survival
sns.pairplot(titanic[['age', 'fare', 'survived']], hue='survived', palette='Set1', diag_kind='kde')
plt.suptitle('Pairplot: Age, Fare, and Survival', fontsize=16)
plt.tight_layout()
plt.show()

# 3. Correlation Heatmap
numeric_df = titanic.select_dtypes(include=['float64', 'int64'])
corr = numeric_df.corr()
plt.figure(figsize=(8, 6))
sns.heatmap(corr, annot=True, cmap='coolwarm', vmin=-1, vmax=1, linewidths=0.5)
plt.title('Correlation Heatmap of Titanic Dataset')
plt.tight_layout()
plt.show()

# 4. Boxplot of Fare by Survival
sns.boxplot(x='survived', y='fare', data=titanic)
plt.title('Fare Distribution by Survival')
plt.xlabel('Survived (0 = No, 1 = Yes)')
plt.ylabel('Fare')
plt.show()
